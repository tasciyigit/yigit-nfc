Python Şifre Kırma Simülasyonu

Amaç

Kullanıcının belirlediği şifreyi, olası tüm kombinasyonları (en düşük karakterden en yükseğe doğru) hızlıca deneyerek bulmaya çalışan bir **Brute-Force algoritmasının** çalışma mantığını göstermek. Deneme hızı, kullanılan bilgisayarın işlem gücüne bağlıdır.

Video

Çalışma Mantığı

Bu projede, Python'da `random` kütüphanesi kullanılarak deneme karakterleri (harf ve rakamlar) tanımlanmıştır. Kod, belirlenen şifre uzunluğunda rastgele karakterler üretir ve şifre bulunana kadar denemeye devam eder. Bu simülasyon, kısa ve zayıf şifrelerin ne kadar hızlı bulunabileceğini göstermektedir.

Kaynak Kod (Python)


# BURAYI DEĞİŞTİR: Python kodunu buraya yapıştır.
from random import randint

KEYS = [
    "1", "2", "3", "4", "5", "6", "7", "8", "9", "0",
    "a", "b", "c", "d", "e", "f", "g", "h", "i", "j",
    "k", "l", "m", "n", "o", "p", "q", "r", "s", "t",
    "u", "v", "w", "x", "y", "z"
]


def brute_force(password: str) -> None:
    pwg = ""
    while pwg != password:
        pwg = ""
        for _ in range(len(password)):
            guess_pass = KEYS[randint(0, len(KEYS) - 1)]
            pwg = f"{guess_pass}{pwg}"
            print(f"\rattacking... please wait! {pwg}", end="", flush=True)
    print()  # newline after carriage-return loop
    print(f"The pass is: {pwg}")


def main() -> None:
    while True:
        password = input("Send the password (max 4 chars): ")
        if len(password) == 0:
            print("Boş şifre kabul edilmiyor.")
            continue
        if len(password) > 4:
            print("En fazla 4 karakter deneyebilirsin. Lütfen yeniden dene.")
            continue
        brute_force(password)
        again = input("Tekrar denemek ister misin? (yes/no): ").strip().lower()
        if again not in ("y", "yes"):
            print("Görüşürüz!")
            break


if __name__ == "__main__":
    main()

Şifreli bomba

Amaç

Bu proje, Keypad (Tuş Takımı), LCD Ekran ve Arduino Uno kullanılarak geliştirilmiş, **kompleks bir şifreli imha simülasyonudur** (Bomb Defusal Game). Projenin amacı, kullanıcının belirlediği süre ve şifre ile kurulan zamanlayıcıyı devreye sokmak ve geri sayım bitmeden doğru şifreyi girerek sistemi durdurmayı sağlamaktır.

Proje, bir Arduino projesinde sıklıkla kullanılan birden fazla temel fonksiyonu bir araya getirir: **Gelişmiş Keypad okuma**, **LCD Ekran üzerinden kullanıcı etkileşimi (kurulum)**, **zamanlayıcı yönetimi (`millis()`)** ve **sesli/görsel geri bildirimler (LED/Buzzer)**.

Yanlış şifre girildiğinde süre kısaltılır ve uyarılar verilirken, süre dolduğunda ise görsel ve sesli **'Patlama Efekti'** ile oyun sonlanır. Bu, gerçek dünya kilit sistemlerinin veya oyun tasarımlarının temel mantığını sergiler.

Devre Şeması

Potansiyometrenin orta pini **Arduino A0** analog girişine, yan pinleri **5V ve GND**'ye bağlanmıştır. Servo Motor'un veri (sarı) pini **Arduino Dijital 9**'a, güç ve toprak pinleri uygun şekilde bağlanmıştır.

Kaynak Kod (C++)


#include 
#include 
#include 

// --- AYARLAR ---
LiquidCrystal_I2C lcd(0x27, 16, 2); 
const int BUZZER = 11; 
const int KIRMIZI_LED = 12;
const int YESIL_LED = 10;   

// --- KEYPAD AYARLARI (Klavye Tersliği Düzeltilmiş) ---
const byte SATIR = 4; 
const byte SUTUN = 4; 
byte satirPinleri[SATIR] = {2, 3, 4, 5}; 
byte sutunPinleri[SUTUN] = {6, 7, 8, 9}; 
char tuslar[SATIR][SUTUN] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
Keypad keypad = Keypad( makeKeymap(tuslar), satirPinleri, sutunPinleri, SATIR, SUTUN );

// --- DEĞİŞKENLER ---
String kuruluSifre = "";
String girilenSifre = "";
long kalanSureSaniye = 0;
unsigned long sonZaman = 0;
bool oyunAktif = false;
bool patladi = false;
bool imhaOldu = false;


// YENİ FONKSİYON: BAŞLANGIÇTA BEKLETME
void ekranBeklet(String ust, String alt, char bekleTusu) {
    lcd.clear();
    lcd.print(ust);
    lcd.setCursor(0, 1);
    lcd.print(alt);
    
    // Belirtilen tuşa basana kadar bekle
    while (keypad.getKey() != bekleTusu) {
        delay(100); 
    }
    tone(BUZZER, 3500, 50); // Tuşa basıldığında bip sesi
}


void setup() {
  lcd.init();
  lcd.backlight();
  pinMode(BUZZER, OUTPUT);
  pinMode(KIRMIZI_LED, OUTPUT); 
  pinMode(YESIL_LED, OUTPUT);
  
  digitalWrite(KIRMIZI_LED, LOW);
  digitalWrite(YESIL_LED, LOW);
  
  // 1. ADIM: BAŞLANGIÇ EKRANI VE BEKLEME
  ekranBeklet("BOMBA SISTEMI ", "* TUSUNA BASIN", '*');
  
  // 2. ADIM: Şifre ve Süre Kurulumu
  sifreBelirleEkrani();
  sureBelirleEkrani();
  
  // 3. ADIM: Oyunu Başlat
  oyunBaslat();
}

void loop() {
  if (!oyunAktif) return; 

  if (millis() - sonZaman >= 1000) {
    sonZaman = millis();
    if (kalanSureSaniye > 0) {
      kalanSureSaniye--;
      ekraniGuncelle();
      tone(BUZZER, 2000, 50); 
    } else {
      patlamaEfekti();
    }
  }

  char tus = keypad.getKey();
  if (tus) {
    tone(BUZZER, 3500, 50); 
    
    if (tus == '#') { 
      if (girilenSifre == kuruluSifre) {
        imhaEfekti();
      } else {
        hataliSifreEfekti();
      }
      girilenSifre = ""; 
      ekraniGuncelle();
    } 
    else if (tus == '*') {
      girilenSifre = "";
      ekraniGuncelle();
    }
    else {
      if(girilenSifre.length() < 8) {
        girilenSifre += tus;
        lcd.setCursor(0, 1);
        lcd.print(girilenSifre);
      }
    }
  }
}

// --- EK FONSİYONLAR ---

// GÜNCELLENMİŞ EKRAN GÜNCELLEME FONKSİYONU (Hata Çözümü)
void ekraniGuncelle() {
  int dk = kalanSureSaniye / 60;
  int sn = kalanSureSaniye % 60;
  
  // Hata Çözümü: Önceki harfleri silmek için sonuna boşluk ekleniyor.
  lcd.setCursor(0, 0);
  lcd.print("SURE: ");
  if (dk < 10) lcd.print("0");
  lcd.print(dk);
  lcd.print(":");
  if (sn < 10) lcd.print("0");
  lcd.print(sn);
  lcd.print("   "); // <-- Bu boşluklar, 'e' harfini veya fazlalıkları temizler.
  
  lcd.setCursor(0, 1);
  if (girilenSifre == "") {
    lcd.print("SIFRE GIRIN...  "); // Alt satırda da temizleme yapalım
  } else {
    lcd.print(girilenSifre);
    lcd.print("                "); // Şifre kısa kalırsa diye temizleme
  }
}

void hataliSifreEfekti() {
  lcd.clear();
  lcd.print("HATALI SIFRE!");
  
  for(int i = 0; i < 20; i++) {
    digitalWrite(KIRMIZI_LED, HIGH);
    digitalWrite(YESIL_LED, LOW);
    tone(BUZZER, 500, 50); 
    delay(50);
    digitalWrite(KIRMIZI_LED, LOW);
    digitalWrite(YESIL_LED, HIGH);
    tone(BUZZER, 100, 50);
    delay(50);
  }
  
  digitalWrite(YESIL_LED, LOW);
  digitalWrite(KIRMIZI_LED, HIGH);
  
  if (kalanSureSaniye > 10) kalanSureSaniye -= 10;
  else kalanSureSaniye = 1; 
}

void oyunBaslat() {
  lcd.clear();
  lcd.print("BOMBA KURULDU!");
  digitalWrite(KIRMIZI_LED, HIGH); 
  digitalWrite(YESIL_LED, LOW);
  delay(2000);
  lcd.clear();
  oyunAktif = true;
  sonZaman = millis();
  ekraniGuncelle();
}

void patlamaEfekti() {
  oyunAktif = false;
  patladi = true;
  lcd.clear();
  lcd.setCursor(4, 0);
  lcd.print("B O O M !");
  lcd.setCursor(2, 1);
  lcd.print("OYUN BITTI");
  
  digitalWrite(YESIL_LED, LOW);
  for(int i = 0; i < 6; i++) {
      digitalWrite(KIRMIZI_LED, HIGH); delay(50);
      digitalWrite(KIRMIZI_LED, LOW); delay(50);
  }
  
  tone(BUZZER, 100); 
  while(true); 
}

void imhaEfekti() {
  oyunAktif = false;
  imhaOldu = true;
  lcd.clear();
  lcd.setCursor(2, 0);
  lcd.print("IMHA EDILDI");
  lcd.setCursor(4, 1);
  lcd.print("BASARILI");
  
  digitalWrite(KIRMIZI_LED, LOW);
  digitalWrite(YESIL_LED, HIGH); 
  
  tone(BUZZER, 1000, 100); delay(150);
  tone(BUZZER, 1200, 100); delay(150);
  tone(BUZZER, 1500, 200); delay(200);
  while(true); 
}

void sifreBelirleEkrani() {
  lcd.clear();
  lcd.print("SIFREYI KURUN:");
  lcd.setCursor(0, 1);
  
  while (true) {
    char key = keypad.getKey();
    if (key) {
      tone(BUZZER, 3000, 50);
      if (key == '#') { if (kuruluSifre.length() > 0) break; } 
      else if (key == '*') { 
        kuruluSifre = "";
        lcd.setCursor(0, 1); lcd.print("                "); lcd.setCursor(0, 1);
      }
      else {
        kuruluSifre += key;
        lcd.print(key);
      }
    }
  }
}

void sureBelirleEkrani() {
  lcd.clear();
  lcd.print("SURE (DAK:SN):");
  lcd.setCursor(0, 1);
  lcd.print("00:00");
  
  String zamanStr = "";
  
  while (true) {
    char key = keypad.getKey();
    if (key && isDigit(key)) {
      tone(BUZZER, 3000, 50);
      
      if (zamanStr.length() < 4) {
        zamanStr += key;
        
        String paddedStr = "0000" + zamanStr;
        paddedStr = paddedStr.substring(paddedStr.length() - 4); 
        
        lcd.setCursor(0, 1);
        lcd.print(paddedStr.substring(0, 2) + ":" + paddedStr.substring(2));
      }
    }
    else if (key == '#') { 
      if (zamanStr.length() > 0) {
        
        String paddedStr = "0000" + zamanStr;
        paddedStr = paddedStr.substring(paddedStr.length() - 4); 
        
        int dakika = paddedStr.substring(0, 2).toInt();
        int saniye = paddedStr.substring(2).toInt();
        
        kalanSureSaniye = (long)(dakika * 60) + saniye; 
        break;
      }
    }
    else if (key == '*') { 
       zamanStr = "";
       lcd.setCursor(0, 1);
       lcd.print("00:00");
    }
  }
}
